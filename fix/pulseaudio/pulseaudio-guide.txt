step1: change "daemonize" from no to yes in /etc/pulse/daemon.conf
step2 : mv pulseaudio.desktop to $HOME/.config/autostart/
step3: mv pulseaudio.service to /etc/systemd/system/
step4: "systemctl enable pulseaudio && systemctl start pulseaudio"
step5: "systemctl start pulseaudio"
